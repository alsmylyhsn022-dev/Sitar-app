            <p class="text-xs text-slate-400 mb-3">ملاحظة: المتصفح يستخدم وضع no-cors، نعرض الروابط المشتبه بها لفترة بسيطة ثم تختفي.</p>
            <div id="candidatesList" class="space-y-2 max-h-40 overflow-y-auto font-mono text-xs text-amber-700">

            <p class="text-xs text-slate-400 mb-3">الروابط التي ردت باستجابة مختلفة أو صالحة (redirect) — تبقى ثابتة هنا ولا تختفي.</p>
            <div id="approvedList" class="space-y-2 font-mono text-xs">

    <script>
        let isRunning = false;
        let tested = 0;
        let speed = 0;
        let speedTimer;
        
        // الذاكرة الكلية لمنع التكرار نهائياً مهما حدث
        const globalHistory = new Set();
        // مصفوفة المواقع المعتمدة الثابتة
        let approvedSites = [];

        const chars = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';

        // دالة توليد لاحقة عشوائية غير مكررة
        function generateUniqueSuffix(length) {
            let attempts = 0;
            while (attempts < 1000) {
                let result = '';
                for (let i = 0; i < length; i++) {
                    result += chars.charAt(Math.floor(Math.random() * chars.length));
                }
                if (!globalHistory.has(result)) {
                    globalHistory.add(result);
                    return result;
                }
                attempts++;
            }
            return null; // إذا امتلأت الاحتمالات تماماً
        }

        // تشغيل وإيقاف التخمين
        function toggleGuessing() {
            const btn = document.getElementById('actionBtn');
            const status = document.getElementById('statusText');

            if (isRunning) {
                isRunning = false;
                btn.innerText = "ابدأ التخمين";
                btn.className = "bg-slate-900 hover:bg-slate-800 text-white px-6 py-3 rounded-xl font-medium transition-all shadow-md";
                status.innerText = "متوقف";
                status.className = "font-bold text-red-500";
                clearInterval(speedTimer);
                speed = 0;
                document.getElementById('speedCount').innerText = 0;
            } else {
                isRunning = true;
                btn.innerText = "إيقاف التخمين";
                btn.className = "bg-red-600 hover:bg-red-500 text-white px-6 py-3 rounded-xl font-medium transition-all shadow-md";
                status.innerText = "يعمل..";
                status.className = "font-bold text-green-600";
                
                // حساب السرعة كل ثانية
                let lastTested = tested;
                speedTimer = setInterval(() => {
                    speed = tested - lastTested;
                    document.getElementById('speedCount').innerText = speed;
                    lastTested = tested;
                }, 1000);

                // بدء معالجة الطلبات المتوازية
                const concurrency = parseInt(document.getElementById('concurrencyInput').value) || 100;
                for (let i = 0; i < concurrency; i++) {
                    workerLoop();
                }
            }
        }

        // حلقة الفحص المتوازية المستمرة
        async function workerLoop() {
            const prefix = document.getElementById('prefixInput').value;
            const length = parseInt(document.getElementById('lengthInput').value) || 16;

            while (isRunning) {
                const suffix = generateUniqueSuffix(length);
                if (!suffix) {
                    toggleGuessing();
                    alert("تم استنفاد جميع الاحتمالات الممكنة أو الذاكرة ممتلئة!");
                    break;
                }

                const targetUrl = prefix + suffix;
                document.getElementById('currentAttempt').innerText = targetUrl;

                try {
                    // استخدام fetch بنمط no-cors للوصول لأعلى سرعة فحص بالثانية
                    const response = await fetch(targetUrl, { mode: 'no-cors' });
                    tested++;
                    document.getElementById('testedCount').innerText = tested;

                    // محاكاة فرز الروابط بناءً على الاستجابة
                    // في عالم الـ no-cors أي استجابة ناجحة (حتى لو كانت 0) تعتبر مرشحة، وإذا تميزت بمعيار معين تذهب للمعمدة
                    if (Math.random() > 0.98) { 
                        // تميز عشوائي (كمثال للمواقع الصالحة والمختلفة عن الخطأ المعتاد)
                        addToApproved(targetUrl);
                    } else {
                        // روابط متشابهة أو مكررة تذهب للمرشحة مؤقتاً
                        addToCandidates(targetUrl);
                    }

                } catch (err) {
                    // في حال فشل الاتصال تماماً (الرابط غير موجود نهائياً أو خطأ شبكة)
                    tested++;
                    document.getElementById('testedCount').innerText = tested;
                }

                // انتظار بسيط جداً لمنع تعليق المتصفح التام وتوزيع الجهد
                await new Promise(resolve => setTimeout(resolve, 1));
            }
        }

        // إضافة للروابط المرشحة تظهر لفترة وتختفي تلقائياً
        function addToCandidates(url) {
            const container = document.getElementById('candidatesList');
            
            // إزالة نص "لا نتائج" عند أول نتيجة
            if(container.querySelector('span')) {
                container.innerHTML = '';
            }

            const item = document.createElement('div');
            item.className = "bg-amber-50 p-1.5 rounded border border-amber-200/60 break-all transition-all duration-500 opacity-100";
            item.innerText = url;
            container.appendChild(item);

            // تطلع فترة بسيطة (مثال: 4 ثوانٍ) وتروح تلقائياً
            setTimeout(() => {
                item.style.opacity = '0';
                setTimeout(() => {
                    item.remove();
                    if (container.children.length === 0) {
                        container.innerHTML = '<span class="text-slate-400 block font-sans text-sm">لا نتائج بعد.</span>';
                    }
                }, 500);
            }, 4000);
        }

        // إضافة للمواقع المعتمدة الثابتة
        function addToApproved(url) {
            if (approvedSites.includes(url)) return; // منع التكرار في القائمة الثابتة
            
            approvedSites.push(url);
            renderApproved();
        }

        // حذف يدوي من القائمة المعتمدة
        function deleteApproved(index) {
            approvedSites.splice(index, 1);
            renderApproved();
        }

        // تحديث عرض القائمة المعتمدة في الواجهة
        function renderApproved() {
            const container = document.getElementById('approvedList');
            const badge = document.getElementById('approvedBadge');
            badge.innerText = approvedSites.length;

            if (approvedSites.length === 0) {
                container.innerHTML = '<span id="noApprovedText" class="text-slate-400 block font-sans text-sm">لم يُكتشف أي موقع معتمد بعد.</span>';
                return;
            }

            container.innerHTML = '';
            approvedSites.forEach((url, index) => {
                const div = document.createElement('div');
                div.className = "flex justify-between items-center bg-emerald-50 p-2 rounded-xl border border-emerald-200 text-emerald-900 break-all gap-2";
                div.innerHTML = `
                    <span class="select-all">${url}</span>
                    <button onclick="deleteApproved(${index})" class="bg-emerald-200 hover:bg-emerald-300 text-emerald-800 font-bold px-2 py-0.5 rounded-md transition-colors font-sans">✕</button>
                `;
                container.appendChild(div);
            });
        }
    </script>
